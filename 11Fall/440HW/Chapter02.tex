MATH 440: Chapter 2 Write-Up Problems\vspace{1em}

Name:\vspace{1em}
\begin{enumerate}

\item Prove or disprove: If $\gcd(a,r)=c$ and $\gcd(b,r)=d$, then $\gcd(ab,r)=cd$.
%155.C53.25.17

\item Give, with proof, a closed formula (i.e. a formula without ``$\cdots$'') for $$1 + x + x^2 + x^3 + \cdots + x^{n-1} + \frac{x^n}{x-1}$$ for any $n \ge 1$.
%155.C53.9.4

\item Prove that if $a^n \mid b^n$ then $a \mid b$. (Hint: Set $d=\gcd(a,b)$ and write $a=rd$ and $b=sd$, where $\gcd(r,s)=1$. Then you can use without proof that if $\gcd(r,s)=1$, then $\gcd(r^n,s^n)=1$. Then show that $r=1$.)
%162.B87.42.11

\item Find, with proof, all integers $x$ such that $3x+7$ is divisible by 11.
%162.B4.15.19

\end{enumerate}
